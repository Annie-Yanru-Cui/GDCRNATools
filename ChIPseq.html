1.2 Trim low-quality reads and remove adaptor sequences

java -jar $Trimmomatic/trimmomatic-0.36.jar SE \
        -threads 8 \
        -phred33 \
        -trimlog trim.log \
        $FILE \
        $SAMPLE.trimmed.fastq.gz \
        ILLUMINACLIP:$adaptor:2:40:15 \
        LEADING:0 \
        TRAILING:0 \
        SLIDINGWINDOW:4:15 \
        MINLEN:50

2 Alignment

2.1 BWA alignment

bwa mem -t $CPU -v 1 -M $genome $FILE > $SAMPLE.sam

2.2 Sort .bam files and remove PCR duplications

samtools sort -@ $CPU -T $SAMPLE.ALIGNtmp -O bam -o $SAMPLE.sorted.bam $SAMPLE.sam

java -Xmx48g -jar $Picard MarkDuplicates \
      ASSUME_SORTED=TRUE \
      REMOVE_DUPLICATES=TRUE \
      VALIDATION_STRINGENCY=LENIENT \
      INPUT=$SAMPLE.sorted.bam \
      OUTPUT=$SAMPLE.sorted.nodup.bam \
      METRICS_FILE=$SAMPLE.nodup.metrics


samtools flagstat $SAMPLE.sorted.nodup.bam > $SAMPLE.sorted.nodup.bam.flagstat
samtools index $SAMPLE.sorted.nodup.bam
